一个用于对任务进行分组并且提供任务执行位置跟踪的工具软件包。
该工具包中主要提供了Task和LocationTask两个工具类，Task是所有任务的最高抽象。而LocationTask是一个可显示执行步骤的任务类型 
import java.lang.Thread.State;
import org.more.task.task.StaticTaskSet;
import org.more.task.task.WaitTask;
public class Test {
    public static void main(String[] args) throws Exception {
        StaticTaskSet t_1 = new StaticTaskSet();
        t_1.addTaskItem(new WaitTask(500));//33
        t_1.addTaskItem(new WaitTask(500));//66
        t_1.addTaskItem(new WaitTask(500));//100
        StaticTaskSet t_2 = new StaticTaskSet();
        t_2.addTaskItem(new WaitTask(500));//10
        t_2.addTaskItem(new WaitTask(500));//20
        t_2.addTaskItem(t_1);//30
        t_2.addTaskItem(new WaitTask(500));//40
        t_2.addTaskItem(new WaitTask(500));//50
        t_2.addTaskItem(new WaitTask(500));//60
        t_2.addTaskItem(new WaitTask(500));//70
        t_2.addTaskItem(new WaitTask(500));//80
        t_2.addTaskItem(new WaitTask(500));//90
        t_2.addTaskItem(new WaitTask(500));//100
        t_2.setShowChildTaskLocation(true);
        Thread t = new Thread(t_2);
        t.start();
        while (true) {
            Thread.sleep(200);
            t_2.updateCurrent();
            System.out.println(((int) (t_2.getRatio() * 100)));
            if (t.getState() == State.TERMINATED)
                break;
        }
    }
}
输出结果是：0，33，33，66，66，66，30，30，40，40，40，50，50，60，60，60，70，70，80，80，80，90，90，100
